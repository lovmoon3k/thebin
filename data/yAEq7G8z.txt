    if not NewbDevBoxDisplayButtonsList then
        NewbDevBoxDisplayButtonsList = {
            [1] = {LogoutButton = true},
            [2] = {ExitButton = true},
            [3] = {FStackButton = true},
            [4] = {ClearChatButton = true},
            [5] = {ChatSpacerButton = true},
            [6] = {ETraceButton = true},
            [7] = {ETraceStartButton = true},
            [8] = {ETraceStopButton = true},
            [9] = {ChatLogButton = true},
            [10] = {CombatLogButton = true},
            [11] = {VersionButton = true},
            [12] = {BugSackButton = false},
            [13] = {BugSackClearButton = false},
            [14] = {WoWLuaButton = false},
            [15] = {REHackButton = false},
            [16] = {ViragDevToolButton = false},
            [17] = {APIInterfaceButton = false},
            [18] = {HandleItButton = false},
            [19] = {WhatsOnTheMapButton = false},
            [20] = {CubeCodeButton = false},
            [21] = {CubeBrowserButton = false},
            [22] = {AddonUsageButton = false}
        }
    end
    if not NewbDevBoxOptionsTable then
        NewbDevBoxOptionsTable = {
            ButtonFrameScale = 1,
            ButtonFrameOpacity = 1,
            ButtonIsVisable = true
        }
    end
end

local function setUpNewbDevBoxSortedTable()
    for k in pairs(NewbDevBoxDisplayButtonsList) do -- insert the key of our button table
        table.insert(SortedNewbDevBoxButtonTable, k)
    end
    table.sort(SortedNewbDevBoxButtonTable) -- sort the key values table
end

local function setCheckBoxesAndButtonFrameMetrics() -- set up the checkboxes and the frame and buttons scale
    for index = 1, #SortedNewbDevBoxButtonTable do -- loop through the sorting table
        tableIndex = SortedNewbDevBoxButtonTable[index] -- get the current index
        tableAtIndex = NewbDevBoxDisplayButtonsList[tableIndex] -- get the table held in the button table index
        for k, v in pairs(tableAtIndex) do -- loop through the table at index
            buttonName = _G["NewbDevBoxInterfaceFramecb" .. k] -- this is our current checkbox name
            if v then -- if true
                buttonName:SetChecked(true) -- check checkbox
            end
        end
    end
    NewbDevBoxButtonFrame:SetScale(NewbDevBoxOptionsTable.ButtonFrameScale) -- set the frame scale
    NewbDevBoxInterfaceFrame.ScaleTextBox:SetNumber(NewbDevBoxOptionsTable.ButtonFrameScale * 100) -- get the scale size and display it
    NewbDevBoxInterfaceFrame.ScaleTextBox:SetCursorPosition(0)
    NewbDevBoxButtonFrame:SetAlpha(NewbDevBoxOptionsTable.ButtonFrameOpacity) -- set the frame opacity
    NewbDevBoxInterfaceFrame.OpacityTextBox:SetNumber(NewbDevBoxOptionsTable.ButtonFrameOpacity * 100) -- get the opacity value and display it
    NewbDevBoxInterfaceFrame.OpacityTextBox:SetCursorPosition(0)
    if NewbDevBoxOptionsTable.ButtonIsVisable then -- if the button was visable last time
        NewbDevBoxButtonFrame:Show()
    else
        NewbDevBoxButtonFrame:Hide()
    end
end

local function checkOtherAddonsEnabled()
    for k, v in pairs(ListOfDevAddons) do
        local name, title, _, enabled = GetAddOnInfo(v)
        if name == "BugSack" then
            if enabled then -- if addon is loaded enable checkboxes and remove grey text
                NewbDevBoxInterfaceFramecbBugSackButton.text:SetText(TextStringList.cbBugSackButton)
                NewbDevBoxInterfaceFramecbBugSackButton:Enable()
                NewbDevBoxInterfaceFramecbBugSackClearButton.text:SetText(TextStringList.cbBugSackClearButton)
                NewbDevBoxInterfaceFramecbBugSackClearButton:Enable()
            else -- ensure button variables are set to false
                NewbDevBoxDisplayButtonsList[12].BugSackButton = false
                NewbDevBoxDisplayButtonsList[13].BugSackClearButton = false
            end
        elseif name == "WowLua" then
            if enabled then
                NewbDevBoxInterfaceFramecbWoWLuaButton.text:SetText(TextStringList.cbWoWLuaButton)
                NewbDevBoxInterfaceFramecbWoWLuaButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[14].WoWLuaButton = false
            end
        elseif name == "REHack" then
            if enabled then
                NewbDevBoxInterfaceFramecbREHackButton.text:SetText(TextStringList.cbREHackButton)
                NewbDevBoxInterfaceFramecbREHackButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[15].REHackButton = false
            end
        elseif name == "ViragDevTool" then
            if enabled then
                NewbDevBoxInterfaceFramecbViragDevToolButton.text:SetText(TextStringList.cbViragDevToolButton)
                NewbDevBoxInterfaceFramecbViragDevToolButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[16].ViragDevToolButton = false
            end
        elseif name == "APIInterface" then
            if enabled then
                NewbDevBoxInterfaceFramecbAPIInterfaceButton.text:SetText(TextStringList.cbAPIInterfaceButton)
                NewbDevBoxInterfaceFramecbAPIInterfaceButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[17].APIInterfaceButton = false
            end
        elseif name == "HandleIt" then
            if enabled then
                NewbDevBoxInterfaceFramecbHandleItButton.text:SetText(TextStringList.cbHandleIt)
                NewbDevBoxInterfaceFramecbHandleItButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[18].HandleItButton = false
            end
        elseif name == "WhatsOnTheMap" then
            if enabled then
                NewbDevBoxInterfaceFramecbWhatsOnTheMapButton.text:SetText(TextStringList.cbWhatsOnTheMapButton)
                NewbDevBoxInterfaceFramecbWhatsOnTheMapButton:Enable()
                WhatsOnTheMapDB.empty = true
                C_Timer.After(
                    .3,
                    function()
                        WhatsOnTheMapFrame:Hide()
                    end
                )
            else
                NewbDevBoxDisplayButtonsList[19].WhatsOnTheMapButton = false
            end
        elseif name == "Cube" then
            if enabled then
                NewbDevBoxInterfaceFramecbCubeCodeButton.text:SetText(TextStringList.cbCubeCodeButton)
                NewbDevBoxInterfaceFramecbCubeCodeButton:Enable()
                NewbDevBoxInterfaceFramecbCubeBrowserButton.text:SetText(TextStringList.cbCubeBrowserButton)
                NewbDevBoxInterfaceFramecbCubeBrowserButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[20].CubeCodeButton = false
                NewbDevBoxDisplayButtonsList[21].CubeBrowserButton = false
            end
        elseif name == "AddonUsage" then
            if enabled then
                NewbDevBoxInterfaceFramecbAddonUsageButton.text:SetText(TextStringList.cbAddonUsageButton)
                NewbDevBoxInterfaceFramecbAddonUsageButton:Enable()
            else
                NewbDevBoxDisplayButtonsList[22].AddonUsageButton = false
            end
        end
    end
end

local function buttonFrameButtonLayout()
    local lastRelative, lastOnTop = NewbDevBoxButtonFrameReloadButton, NewbDevBoxButtonFrameReloadButton
    local lastCount = 1
    local buttonFrameHeight = NumberList.buttonFrameDefaultHeight
    NewbDevBoxButtonFrameLogoutButton:Hide() -- hide all buttons to ensure no ghosting
    NewbDevBoxButtonFrameExitButton:Hide()
    NewbDevBoxButtonFrameFStackButton:Hide()
    NewbDevBoxButtonFrameClearChatButton:Hide()
    NewbDevBoxButtonFrameChatSpacerButton:Hide()
    NewbDevBoxButtonFrameETraceButton:Hide()
    NewbDevBoxButtonFrameETraceStartButton:Hide()
    NewbDevBoxButtonFrameETraceStopButton:Hide()
    NewbDevBoxButtonFrameChatLogButton:Hide()
    NewbDevBoxButtonFrameCombatLogButton:Hide()
    NewbDevBoxButtonFrameVersionButton:Hide()
    NewbDevBoxButtonFrameBugSackButton:Hide()
    NewbDevBoxButtonFrameBugSackClearButton:Hide()
    NewbDevBoxButtonFrameWoWLuaButton:Hide()
    NewbDevBoxButtonFrameREHackButton:Hide()
    NewbDevBoxButtonFrameViragDevToolButton:Hide()
    NewbDevBoxButtonFrameAPIInterfaceButton:Hide()
    NewbDevBoxButtonFrameHandleItButton:Hide()
    NewbDevBoxButtonFrameWhatsOnTheMapButton:Hide()
    NewbDevBoxButtonFrameCubeCodeButton:Hide()
    NewbDevBoxButtonFrameCubeBrowserButton:Hide()
    NewbDevBoxButtonFrameAddonUsageButton:Hide()
    NewbDevBoxButtonFrame:SetSize(NumberList.buttonFrameWidth, buttonFrameHeight) -- reset frame size
    for index = 1, #SortedNewbDevBoxButtonTable do -- loop through the sorting table
        tableIndex = SortedNewbDevBoxButtonTable[index] -- get the current index
        tableAtIndex = NewbDevBoxDisplayButtonsList[tableIndex] -- get the table held in the button table index
        for k, v in pairs(tableAtIndex) do -- loop through the table at index
            if v then
                thisButton = _G["NewbDevBoxButtonFrame" .. k] -- set this button to the global button name
                thisButton:ClearAllPoints()
                thisButton:Show()
                if mod(lastCount, 3) == 0 then -- when the button count reaches 3
                    thisButton:SetPoint("TOP", lastOnTop, "BOTTOM", 0, -5) -- set the button point to the last top button
                    lastOnTop = thisButton -- set the current button as the lastOnTop button
                    buttonFrameHeight = buttonFrameHeight + NumberList.buttonFrameHeightExpander -- expand the frame to allow another row of buttons
                    NewbDevBoxButtonFrame:SetSize(NumberList.buttonFrameWidth, buttonFrameHeight) -- set the frame size
                else
                    thisButton:SetPoint("LEFT", lastRelative, "RIGHT", NumberList.actionButtonSideGap, 0) -- set the button point to the last button
                end
                lastCount = lastCount + 1 -- increase the lastCount
                lastRelative = thisButton -- set this button to the lastRelative
            end
        end
    end
end

local function updateButtonFrameScale(updateScaleNumber)
    if updateScaleNumber >= 301 then -- if the entered number is too high
        updateScaleNumber = 300
    elseif updateScaleNumber <= 49 then -- if the entered number is too low
        updateScaleNumber = 50
    end
    NewbDevBoxOptionsTable.ButtonFrameScale = updateScaleNumber / 100
    NewbDevBoxButtonFrame:SetScale(NewbDevBoxOptionsTable.ButtonFrameScale)
    NewbDevBoxInterfaceFrame.ScaleTextBox:SetNumber(updateScaleNumber)
end

local function updateButtonFrameOpacity(updateOpacityNumber)
    if updateOpacityNumber >= 101 then -- if the entered number is too high
        updateOpacityNumber = 100
    elseif updateOpacityNumber <= 24 then -- if the entered number is too low
        updateOpacityNumber = 25
    end
    NewbDevBoxOptionsTable.ButtonFrameOpacity = updateOpacityNumber / 100
    NewbDevBoxButtonFrame:SetAlpha(NewbDevBoxOptionsTable.ButtonFrameOpacity)
    NewbDevBoxInterfaceFrame.OpacityTextBox:SetNumber(updateOpacityNumber)
end

-- ===========================         toggle newb dev box interface frame          ===========================
function NEWBDEVBOXTOGGLENEWBDEVBOXINTERFACEFRAME()
    if SettingsPanel:IsShown() then
        HideUIPanel(SettingsPanel)
    else
        --        HideUIPanel(SettingsPanel)
        --         InterfaceOptionsFrame_OpenToCategory(NewbDevBoxInterfaceFrame) -- have to do this twice because reasons
        InterfaceOptionsFrame_OpenToCategory(NewbDevBoxInterfaceFrame) -- debug --
    end
end

-- ===========================         toggle new dev box button frame          ===========================
function NEWBDEVBOXTOGGLENEWBDEVBOXBUTTONFRAME()
    if NewbDevBoxButtonFrame:IsShown() then
        NewbDevBoxButtonFrame:Hide()
    else
        NewbDevBoxButtonFrame:Show()
    end
end

NewbDevBoxButtonFrame:SetSize(NumberList.buttonFrameWidth, NumberList.buttonFrameDefaultHeight)
NewbDevBoxButtonFrame:ClearAllPoints()
NewbDevBoxButtonFrame:SetPoint("CENTER")
NewbDevBoxButtonFrame:EnableMouse(true)
NewbDevBoxButtonFrame:SetMovable(true)
NewbDevBoxButtonFrame:SetClampedToScreen(true)
NewbDevBoxButtonFrame:RegisterForDrag("RightButton", "LeftButton")
NewbDevBoxButtonFrame:SetScript("OnDragStart", NewbDevBoxButtonFrame.StartMoving)
NewbDevBoxButtonFrame:SetScript("OnDragStop", NewbDevBoxButtonFrame.StopMovingOrSizing)
NewbDevBoxButtonFrame:SetToplevel(true)
NewbDevBoxButtonFrame:SetScript(
    "OnEnter",
    function()
        NewbDevBoxButtonFrame:SetAlpha(1)
    end
)
NewbDevBoxButtonFrame:SetScript(
    "OnLeave",
    function()
        NewbDevBoxButtonFrame:SetAlpha(NewbDevBoxOptionsTable.ButtonFrameOpacity)
    end
)

local NewbDevBoxButtonFrameHandleItButton =
    CreateFrame(
    "Button",
    "NewbDevBoxButtonFrameHandleItButton",
    NewbDevBoxButtonFrame,
    TextDimensionList.buttonTemplate
)
NewbDevBoxButtonFrameHandleItButton:SetSize(NumberList.actionButtonWidth, NumberList.actionButtonHeight)
NewbDevBoxButtonFrameHandleItButton:SetText(TextStringList.btnHandleIt)
NewbDevBoxButtonFrameHandleItButton:SetNormalFontObject(TextDimensionList.buttonFont)
NewbDevBoxButtonFrameHandleItButton:SetHighlightFontObject(TextDimensionList.buttonFont)
NewbDevBoxButtonFrameHandleItButton:SetScript(
    "OnClick",
    function()
        if InterfaceOptionsFrame:IsShown() then
            InterfaceOptionsFrame:Hide()
        else
            InterfaceOptionsFrame:Show()
            InterfaceOptionsFrameTab2:Click() -- click the addons tab
            for k, v in pairs(INTERFACEOPTIONS_ADDONCATEGORIES) do -- loop through all categories
                for a, b in pairs(v) do -- loop through the current category
                    if b == "HandleIt" then -- if the current name value
                        thisButton = _G["InterfaceOptionsFrameAddOnsButton" .. k] -- set the button name with the key value
                        thisButton:Click()
                    end
                end
            end
        end
    end
)
