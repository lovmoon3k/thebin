// Тестовая программа :
// Создать и инициализировать 2 массива A[N1], B[N2]
// Найти положение отрицательного элемента массива.
// Полученный указатель разделяет массив на 2 части.
// Определить максимальный элемент в второй  части массив
// Пункты 2 и 3 выполнить для массивов A и B


const int n = 15;
int i, j;
int f(int x[n]) {
	int pol = 0, count = 0;
	for (j = 0; j < n; j++) {    
		if (x[j] > 0) {
      pol = x[j];
      count = j;
      }
	}
  cout << "Последнее положительное число = " << pol << '\n' << "Его индекс равен = " << count << '\n';
	if (pol > 0) {               
		int* prt_pol = &count;
		return *prt_pol;
	}
	else {
		cout << "положительных элементов нет" << '\n';
    return 0;
	}
}
int main() {
	int a[n], amount = 0;
	for (i = 0; i < n; i++){       // заполнение массива
		a[i] = rand() % 100 - 50;
		cout << setw(4) << a[i];
	}
	cout << '\n';                  
	for (j = f(a); j < n; j++) { 
		if (a[j] < 0) {
			amount += 1;
		}
	}
	cout << "Кол-во отрицательных чисел = "<< amount;                 
}
