#!/usr/bin/python
# -*- coding: utf-8 -*-

''' Aufgabenstellung

    logische Operatoren

'''
    
import sys

# Variablen und Werte festlegen
wertX = 12
wertY = 15
wertZ = 20

# Funktionen werden mit def eingeleitet. 
def destroy():
    sys.exit()

# Hauptprogramm

if __name__ == '__main__':  # Programmstart

    print ('Logische Operatoren AND / OR / NOT')
    try:
        # Ausgabe der Werte
        print ('X: ', wertX)
        print ('Y: ', wertY)
        print ('Z: ', wertZ)
        
        # Bedingung 1
        if wertX < wertY and wertX < wertZ:
            print ('X ist die kleinste Zahl.')
        
        # Bedingung 2
        if wertY > wertX or wertY > wertZ:
            print ('Y ist nicht die kleinste Zahl.')
            
        # Bedingung 3
        if not wertY < wertX:
            print ('Y ist nicht kleiner als X.')
    except KeyboardInterrupt:  # wenn 'CTRL-C' gedrÃ¼ckt, dann Ende
        destroy()
