signed main() {
#ifdef _DEBUG
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif
	ios_base::sync_with_stdio(0);
	cin.tie(NULL);
	cout.tie(NULL);
	ll n1, n2, n3, n4, i, j;
	cin >> n1;
	vector <ll> a1(n1);
	for (i = 0; i < n1; i++) {
		cin >> a1[i];
	}
	cin >> n2;
	vector <ll> a2(n2);
	for (i = 0; i < n2; i++) {
		cin >> a2[i];
	}
	cin >> n3;
	vector <ll> a3(n3);
	for (i = 0; i < n3; i++) {
		cin >> a3[i];
	}
	cin >> n4;
	vector <ll> a4(n4);
	for (i = 0; i < n4; i++) {
		cin >> a4[i];
	}
	sort(a1.begin(), a1.end());
	sort(a2.begin(), a2.end());
	sort(a3.begin(), a3.end());
	sort(a4.begin(), a4.end());
	ll mn = min(min(a1[0], a2[0]), min(a3[0], a4[0])), mx = max(max(a1[0], a2[0]), max(a3[0], a4[0]));
	ll rzn = abs(mx - mn), rzn2;
	ll pos1 = 0, pos2 = 0, pos3 = 0, pos4 = 0;
	ll mn1 = pos1;
	ll mn2 = pos2;
	ll mn3 = pos3;
	ll mn4 = pos4;
	while (pos1 < n1 && pos2 < n2 && pos3 < n3 && pos4 < n4) {
		mn = min(min(a1[pos1], a2[pos2]), min(a3[pos3], a4[pos4]));
		mx = max(max(a1[pos1], a2[pos2]), max(a3[pos3], a4[pos4]));
		rzn2 = abs(mx - mn);
		if (rzn2 < rzn) {
			rzn = rzn2;
			mn1 = pos1;
			mn2 = pos2;
			mn3 = pos3;
			mn4 = pos4;
		}
		if (rzn2 < 1) {
			break;
		}
		while (pos1 < n1 && mn == a1[pos1]) {
			pos1++;
		}
		while (pos2 < n2 && mn == a2[pos2]) {
			pos2++;
		}
		while (pos3 < n3 && mn == a3[pos3]) {
			pos3++;
		}
		while (pos4 < n4 && mn == a4[pos4]) {
			pos4++;
		}
	}
	cout << a1[mn1] << " " << a2[mn2] << " " << a3[mn3] << " " << a4[mn4] << endl;
}
