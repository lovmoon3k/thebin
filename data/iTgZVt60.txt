let = {}
for i in alp:
    let[i] = i


def delt(a, b):
    for i in range(len(a)):
        let[a[i]] = b[i]

def get(s):
    for i in s:
        print(let[i], end = '')
    print()


delt("г", "е")

def see():
    for i in [s1, s2, s3]:
        get(i)
