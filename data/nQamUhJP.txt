input_cars = int(input("В наличии авто: ")) # спрашиваем сколько есть автомобилей
input_slots = int(input("В маишне: ")) # спрашиваем сколько человек может быть в машине
input_drivers = int(input("Водителей: ")) # спрашивает сколько водитилеей мы имеем

print("Пустует автомобилей: %d" % (input_cars - input_drivers)) # выводим число незадействованных автомобилей
print("Можно перевезти: %d" % (input_slots * input_drivers)) # выводим число человек которых можно перевести
print("В каждой машине будет примерно: %d" % ((input_slots * input_drivers / input_drivers) - 1)) # выводим число человек в каждой машине в среднем
