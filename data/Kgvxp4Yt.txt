#include <iostream>
using namespace std;
/*
2023.01.17 zad1
Задача 1
Да се напише програма, която прочита от клавиатурата цяло положително число N,
след което прочита последователно N на брой цели числа и намира и извежда сумата на всички
положителни числа измежду въведените.

Вход
Първият ред съдържа едно цяло положително число N.
Вторият ред съдържа N на брой цели числа.

Изход
Едно число, представляващо сумата на въведените положителни числа.

Пример:
Вход
6
5 8 -2 3 -4 0
Изход
16
*/
int main()
{
    int n, currrent, sum = 0;
    cin >> n;
    for(int i=1; i<=n; i++)
    {
        cin >> currrent;
        if(currrent > 0)
        {
            sum += currrent;
        }
    }


    cout << sum;


    return 0;
}
