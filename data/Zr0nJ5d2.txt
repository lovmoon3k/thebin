% Algorytm Symulowanego wy≈ºarzania (Simulated Annealing)

clear all
close all
clc
clearvars

format long
rng('default') 

myfcn  = @(x,y) sin(x).*exp((1-cos(y)).^2)+cos(y).*exp((1-sin(x)).^2)+(x-y).^2;

lb=[-2*pi -2*pi]
ub=[2*pi 2*pi]
     
figure(1)
fsurf(myfcn ,[lb(1) ub(1) lb(2) ub(2)])
hold on
grid on
