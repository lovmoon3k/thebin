function fibonacciGenerator (n) {
//Do NOT change any of the code above ðŸ‘†
    
    //Write your code here:
    
    
var output = [];

for(var i=0; i < n; i++){
      
    if (i === 0){
        output[i] = 0;
    }
    
    else if (i === 1){
        output[i] = 1;
    }
    
    else{
    output[i] = Number(output[i-1]) + Number(output[i-2]);
    }
  
  
}

return output;
